118 auto ProduitScal = [](const Point2D<T>& P, const Point2D<T>& Q, const Point2D<T>& R) {
119 return (Q.getX() - P.getX()) * (R.getY() - P.getY()) -
120 (Q.getY() - P.getY()) * (R.getX() - P.getX());
